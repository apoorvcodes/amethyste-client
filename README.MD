### JavaScript
```js

const amiapi = require("amethyste-client")

const Client = amiapi("UR_TOKEN")

Client.generate({
	endpoint: "beautiful",
	options: {
		"url": client.user.avatarURL
	}
}).then(img => {
	console.log(img)
})
```


### TypeScript
```ts

import amiapi from "amethyste-client"

const Client:amiapi = amiapi("UR_TOKEN")

Client.generate({
	endpoint: "beautiful",
	options: {
		"url": client.user.avatarURL
	}
  }).then(img => {
	console.log(img)
})
```
